const uint8_t PULSE_PIN = 3;

void setup() {
  pinMode(PULSE_PIN, OUTPUT);
  digitalWrite(PULSE_PIN, LOW);

  Serial.begin(115200);
  while (!Serial) { }   // Required on Arduino R4
}

void loop() {
  if (!Serial.available()) return;

  unsigned long freq     = Serial.parseInt();
  unsigned long pulse_ms = Serial.parseInt();
  unsigned long count    = Serial.parseInt();

  if (freq == 0 || pulse_ms == 0 || count == 0) return;

  unsigned long period_us = 1000000UL / freq;
  unsigned long high_us   = pulse_ms * 1000UL;

  unsigned long next_edge = micros();

  for (unsigned long i = 0; i < count; i++) {
    // HIGH phase
    digitalWrite(PULSE_PIN, HIGH);
    delayMicroseconds(high_us);

    // LOW phase
    digitalWrite(PULSE_PIN, LOW);

    // Wait until next period boundary (anchored timing)
    next_edge += period_us;
    while ((long)(micros() - next_edge) < 0) { }
  }

  Serial.println("DONE");
}
